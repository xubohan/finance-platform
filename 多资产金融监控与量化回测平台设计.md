### 5.2 DataFetcher 基类（熔断器）

```python
# app/services/data\_fetcher/base.py

from dataclasses import dataclass, field
from datetime import datetime, timedelta, timezone
import time
import logging

logger = logging.getLogger(\_\_name\_\_)

@dataclass
class CircuitBreakerState:
    failures:       int      = 0
    last\_failure:   datetime | None = None
    open\_until:     datetime | None = None
    threshold:      int      = 3
    recovery\_sec:   int      = 60

    def record\_failure(self) -> None:
        self.failures += 1
        self.last\_failure = datetime.now(timezone.utc)
        if self.failures >= self.threshold:
            self.open\_until = self.last\_failure + timedelta(seconds=self.recovery\_sec)

    def is\_open(self) -> bool:
        if self.open\_until is None:
            return False
        if datetime.now(timezone.utc) > self.open\_until:
            self.failures = 0
            self.open\_until = None
            return False
        return True

    def record\_success(self) -> None:
        self.failures = 0
        self.open\_until = None


class BaseDataFetcher:
    name: str = "base"

    def \_\_init\_\_(self):
        self.\_cb = CircuitBreakerState()

    def fetch\_ohlcv(self, symbol: str, start: str,
                    end: str, interval: str) -> pd.DataFrame:
        raise NotImplementedError

    def fetch\_quote(self, symbol: str) -> dict:
        raise NotImplementedError

    def fetch\_fundamentals(self, symbol: str,
                           report\_type: str = "income") -> pd.DataFrame:
        raise NotImplementedError

    def \_call(self, func, \*args, \*\*kwargs):
        if self.\_cb.is\_open():
            logger.warning("%s circuit open, skipping", self.name)
            raise RuntimeError(f"{self.name} circuit breaker open")
        try:
            result = func(\*args, \*\*kwargs)
            self.\_cb.record\_success()
            return result
        except Exception as exc:
            self.\_cb.record\_failure()
            logger.warning("%s call failed: %s", self.name, exc)
            raise
```

# app/services/data\_fetcher/akshare\_adapter.py

import akshare as ak

class AKShareAdapter(BaseDataFetcher):
    name = "akshare"

    def fetch\_ohlcv(self, symbol, start, end, interval="daily") -> pd.DataFrame:
        """
        symbol格式: 600000.SH / 000001.SZ

        支持interval: daily/weekly/monthly/60/30/15/5/1

### 5.4 LLM 服务（OpenAI-compatible）

```python
# app/services/llm\_service.py

from openai import AsyncOpenAI
import json

class LLMService:
    def \_\_init\_\_(self, settings):
        self.client = AsyncOpenAI(
            api\_key=settings.llm\_api\_key,
            base\_url=settings.llm\_base\_url,
        )
        self.model = settings.llm\_model

    async def analyze\_news(self, title: str, content: str,
                           symbols: list\[str]) -> dict:
        """情绪分析 + 摘要，用于批量处理新闻"""
        prompt = f"""分析以下金融新闻，以JSON格式返回（仅JSON，无其他内容）：

标题：{title}
内容：{content\[:1000]}
关联标的：{', '.join(symbols) if symbols else '未知'}

返回格式：
{{
  "sentiment": 0.72,
  "importance": 3,
  "summary": "50字内核心摘要",
  "impact\_assessment": "对关联标的影响判断",
  "key\_factors": \["因素1", "因素2"],
  "category": "macro"
}}

category枚举: earnings|macro|policy|geopolitical|social|other
sentiment范围: -1.0(极负面) \~ 1.0(极正面)"""

        resp = await self.client.chat.completions.create(
            model=self.model,
            messages=\[{"role": "user", "content": prompt}],
            max\_tokens=400,
            response\_format={"type": "json\_object"},
        )
        return json.loads(resp.choices\[0].message.content)

    async def analyze\_event\_impact(
        self,
        event\_text: str,
        event\_type: str,
        historical\_stats: dict,
        target\_symbols: list\[str],
    ) -> dict:
        """深度事件影响分析，结合历史统计数据"""
        stats\_summary = json.dumps(historical\_stats, ensure\_ascii=False)
        prompt = f"""你是量化分析师。基于以下信息，分析该事件对市场的影响：

【事件内容】
{event\_text\[:2000]}

【事件类型】{event\_type}

【历史同类事件统计】
{stats\_summary}

【目标标的】{', '.join(target\_symbols)}

请返回JSON（仅JSON）：
{{
  "sentiment\_score": 0.72,
  "sentiment\_label": "positive",
  "summary": "事件摘要100字内",
  "key\_factors": \["正面因素1", "正面因素2"],
  "risk\_factors": \["风险因素1"],
  "impact\_assessment": "综合影响评估",
  "symbol\_analysis": {{
    "SPY": {{
      "direction": "up",
      "confidence": 0.65,
      "reasoning": "理由"
    }}
  }}
}}"""

        resp = await self.client.chat.completions.create(
            model=self.model,
            messages=\[{"role": "user", "content": prompt}],
            max\_tokens=800,
            response\_format={"type": "json\_object"},
        )
        return json.loads(resp.choices\[0].message.content)

    async def batch\_analyze\_news(
        self, items: list\[dict], concurrency: int = 5
    ) -> list\[dict]:
        """带并发控制的批量处理"""
        import asyncio
        semaphore = asyncio.Semaphore(concurrency)

        async def \_single(item):
            async with semaphore:
                try:
                    return await self.analyze\_news(
                        item\["title"],
                        item.get("content", ""),
                        item.get("symbols", \[]),
                    )
                except Exception as e:
                    logger.error("LLM analysis failed: %s", e)
                    return None

        return await asyncio.gather(\*\[\_single(i) for i in items])
```
